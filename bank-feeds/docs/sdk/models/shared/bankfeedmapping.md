# BankFeedMapping

A bank feed connection between a source account and a target account, including potential target accounts.

## Example Usage

```typescript
import { BankFeedMapping } from "@codat/bank-feeds/sdk/models/shared";
import { Decimal } from "@codat/bank-feeds/sdk/types";

let value: BankFeedMapping = {
  sourceAccountId: "c53fd75a-dddb-4493-839a-5090b008f625",
  sourceAccountName: "bank name here",
  sourceAccountNumber: "5670",
  sourceBalance: "1550",
  sourceCurrency: "GBP",
  status: "pending",
  targetAccountOptions: [
    {
      id: "044096a7-1478-4554-8820-f478caad1b00",
      name: "account-081",
      balance: new Decimal("100.99"),
      sortCode: "602336",
      accountNumber: "23203",
    },
  ],
};
```
