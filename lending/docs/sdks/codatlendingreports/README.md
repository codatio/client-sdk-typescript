  authHeader: "Basic BASE_64_ENCODED(API_KEY)",
});

async function run() {
  const result = await codatLending.accountsReceivable.reports.getAgedDebtors({
    companyId: "8a210b68-6988-11ed-a1eb-0242ac120002",
    reportDate: new RFCDate("2022-12-31"),
    numberOfPeriods: 12,
    periodLengthDays: 30,
  });

  // Handle the result
  console.log(result);
