# CommerceCompanyInfo

In the Codat system, company profile includes standard commercial details about 
a linked company, such as their address, phone number, and company registration.


## Example Usage

```typescript
import { CommerceCompanyInfo } from "@codat/lending/sdk/models/shared";

let value: CommerceCompanyInfo = {
  modifiedDate: "2022-10-23T00:00:00Z",
  sourceModifiedDate: "2022-10-23T00:00:00Z",
  companyName: "Codat",
  companyLegalName: "Codat Limited",
  phoneNumbers: [
    {
      number: "+44 25691 154789",
      type: "Fax",
    },
  ],
  webLinks: [
    {
      type: "Website",
      url: "https://codat.io",
    },
  ],
  registrationNumber: "10480375",
  baseCurrency: "USD",
  accountBalances: [
    {
      currency: "GBP",
    },
  ],
  sourceUrls: {
    "url1": "https://connect.sandbox.com/v2/customers",
    "url2": "https://connect.sandbox.com/v2/disputes",
  },
  createdDate: "2022-10-23T00:00:00Z",
};
```
